JavaScript

            
const incidentData = [
  { time: '08:00', count: 2, risk: 70 },
  { time: '10:00', count: 4, risk: 75 },
  { time: '12:00', count: 6, risk: 82 },
  { time: '14:00', count: 5, risk: 78 },
  { time: '16:00', count: 7, risk: 86 },
  { time: '18:00', count: 4, risk: 74 },
];

function getTodaysAccidentCount(data) {
  return data.reduce((total, item) => total + item.count, 0);
}

function getAverageRisk(data) {
  return Math.round(data.reduce((sum, item) => sum + item.risk, 0) / data.length);
}

function renderAccidentMetrics() {
  const totalIncidents = getTodaysAccidentCount(incidentData);
  const averageRisk = getAverageRisk(incidentData);
  console.log(`Total accidents: ${totalIncidents}`);
  console.log(`Average risk score: ${averageRisk}%`);
}

renderAccidentMetrics();

        


        

            
About the Prediction Model

            
This example uses hourly accident data and risk scoring to simulate a prediction dashboard. The code is structured for HTML layout, dashboard styling, and lightweight JS analytics that could be extended into a real-time system.
